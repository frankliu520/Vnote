### 1.3.1. nst17.c
```
// SPDX-License-Identifier: GPL-2.0-or-later
/*
 * nst175.c - Part of lm_sensors, Linux kernel modules for hardware
 *	 monitoring
 * Copyright (c) 1998, 1999  Frodo Looijaard <frodol@dds.nl>
 * NST175 adaptation
 */

#include <linux/module.h>
#include <linux/init.h>
#include <linux/slab.h>
#include <linux/jiffies.h>
#include <linux/i2c.h>
#include <linux/hwmon.h>
#include <linux/hwmon-sysfs.h>
#include <linux/err.h>
#include <linux/of_device.h>
#include <linux/of.h>
#include <linux/regmap.h>
#include <linux/util_macros.h>
#include <linux/regulator/consumer.h>
#include "nst175.h"

/*
 * This driver handles the NST175 and compatible digital temperature sensors.
 */

enum nst175_type {
	nst175,
};

/**
 * struct nst175_params - nst175 configuration parameters.
 * @set_mask:		Bits to set in configuration register when configuring
 *			the chip.
 * @clr_mask:		Bits to clear in configuration register when configuring
 *			the chip.
 * @default_resolution:	Default number of bits to represent the temperature
 *			value.
 * @resolution_limits:	Limit register resolution. Optional. Should be set if
 *			the resolution of limit registers does not match the
 *			resolution of the temperature register.
 * @resolutions:	List of resolutions associated with sample times.
 *			Optional. Should be set if num_sample_times is larger
 *			than 1, and if the resolution changes with sample times.
 *			If set, number of entries must match num_sample_times.
 * @default_sample_time:Sample time to be set by default.
 * @num_sample_times:	Number of possible sample times to be set. Optional.
 *			Should be set if the number of sample times is larger
 *			than one.
 * @sample_times:	All the possible sample times to be set. Mandatory if
 *			num_sample_times is larger than 1. If set, number of
 *			entries must match num_sample_times.
 */

struct nst175_params {
	u8			set_mask;
	u8			clr_mask;
	u8			default_resolution;
	u8			resolution_limits;
	const u8		*resolutions;
	unsigned int		default_sample_time;
	u8			num_sample_times;
	const unsigned int	*sample_times;
};

/* Addresses scanned */
static const unsigned short normal_i2c[] = { 0x48, 0x49, 0x4a, 0x4b, 0x4c,
					0x4d, 0x4e, 0x4f, I2C_CLIENT_END };

/* The NST175 registers */
#define NST175_REG_TEMP		0x00
#define NST175_REG_CONF		0x01
#define NST175_REG_HYST		0x02
#define NST175_REG_MAX		0x03

/* Each client has this additional data */
struct nst175_data {
	struct i2c_client		*client;
	struct regmap			*regmap;
	struct regulator		*vs;
	u8				orig_conf;
	u8				current_conf;
	u8				resolution;	/* In bits, 9 to 16 */
	unsigned int			sample_time;	/* In ms */
	enum nst175_type		kind;
	const struct nst175_params	*params;
};

/*-----------------------------------------------------------------------*/

static const u8 nst175_sample_set_masks[] = { 0 << 5, 1 << 5, 2 << 5, 3 << 5 };

#define NST175_SAMPLE_CLEAR_MASK	(3 << 5)

/* The structure below stores the configuration values of the supported devices.
 * In case of being supported multiple configurations, the default one must
 * always be the first element of the array
 */
static const struct nst175_params device_params[] = {
	[nst175] = {
		.clr_mask = NST175_SHUTDOWN,	/* Ensure continuous conversion */
		.default_resolution = 12,
		.default_sample_time = 100,
		/* Add more specific parameters for NST175 as needed */
	},
};

static inline long nst175_reg_to_mc(s16 temp, u8 resolution)
{
	return ((temp >> (16 - resolution)) * 1000) >> (resolution - 8);
}

static int nst175_write_config(struct nst175_data *data, u8 set_mask,
			     u8 clr_mask)
{
	u8 value;

	clr_mask |= NST175_SHUTDOWN;
	value = data->current_conf & ~clr_mask;
	value |= set_mask;

	if (data->current_conf != value) {
		s32 err;

		err = i2c_smbus_write_byte_data(data->client, NST175_REG_CONF,
						value);
		if (err)
			return err;
		data->current_conf = value;
	}
	return 0;
}

static int nst175_read(struct device *dev, enum hwmon_sensor_types type,
		     u32 attr, int channel, long *val)
{
	struct nst175_data *data = dev_get_drvdata(dev);
	unsigned int regval;
	int err, reg;

	switch (type) {
	case hwmon_chip:
		switch (attr) {
		case hwmon_chip_update_interval:
			*val = data->sample_time;
			break;
		default:
			return -EINVAL;
		}
		break;
	case hwmon_temp:
		switch (attr) {
		case hwmon_temp_input:
			reg = NST175_REG_TEMP;
			break;
		case hwmon_temp_max:
			reg = NST175_REG_MAX;
			break;
		case hwmon_temp_max_hyst:
			reg = NST175_REG_HYST;
			break;
		default:
			return -EINVAL;
		}
		err = regmap_read(data->regmap, reg, &regval);
		if (err < 0)
			return err;

		*val = nst175_reg_to_mc(regval, data->resolution);
		break;
	default:
		return -EINVAL;
	}
	return 0;
}

static int nst175_write_temp(struct device *dev, u32 attr, long temp)
{
	struct nst175_data *data = dev_get_drvdata(dev);
	u8 resolution;
	int reg;

	switch (attr) {
	case hwmon_temp_max:
		reg = NST175_REG_MAX;
		break;
	case hwmon_temp_max_hyst:
		reg = NST175_REG_HYST;
		break;
	default:
		return -EINVAL;
	}

	/*
	 * Resolution of limit registers is assumed to be the same as the
	 * temperature input register resolution unless given explicitly.
	 */
	if (data->params->resolution_limits)
		resolution = data->params->resolution_limits;
	else
		resolution = data->resolution;

	temp = clamp_val(temp, NST175_TEMP_MIN, NST175_TEMP_MAX);
	temp = DIV_ROUND_CLOSEST(temp  << (resolution - 8),
				 1000) << (16 - resolution);

	return regmap_write(data->regmap, reg, (u16)temp);
}

static int nst175_update_interval(struct device *dev, long val)
{
	struct nst175_data *data = dev_get_drvdata(dev);
	u8 index;
	s32 err;

	if (data->params->num_sample_times == 0)
		return -EOPNOTSUPP;

	index = find_closest(val, data->params->sample_times,
			     (int)data->params->num_sample_times);

	err = nst175_write_config(data, nst175_sample_set_masks[index],
				NST175_SAMPLE_CLEAR_MASK);
	if (err)
		return err;

	data->sample_time = data->params->sample_times[index];
	if (data->params->resolutions)
		data->resolution = data->params->resolutions[index];

	return 0;
}

static int nst175_write_chip(struct device *dev, u32 attr, long val)
{
	switch (attr) {
	case hwmon_chip_update_interval:
		return nst175_update_interval(dev, val);
	default:
		return -EINVAL;
	}
	return 0;
}

static int nst175_write(struct device *dev, enum hwmon_sensor_types type,
		      u32 attr, int channel, long val)
{
	switch (type) {
	case hwmon_chip:
		return nst175_write_chip(dev, attr, val);
	case hwmon_temp:
		return nst175_write_temp(dev, attr, val);
	default:
		return -EINVAL;
	}
	return 0;
}

static umode_t nst175_is_visible(const void *data, enum hwmon_sensor_types type,
			       u32 attr, int channel)
{
	const struct nst175_data *config_data = data;

	switch (type) {
	case hwmon_chip:
		switch (attr) {
		case hwmon_chip_update_interval:
			if (config_data->params->num_sample_times > 1)
				return 0644;
			return 0444;
		}
		break;
	case hwmon_temp:
		switch (attr) {
		case hwmon_temp_input:
			return 0444;
		case hwmon_temp_max:
		case hwmon_temp_max_hyst:
			return 0644;
		}
		break;
	default:
		break;
	}
	return 0;
}

static const struct hwmon_channel_info *nst175_info[] = {
	HWMON_CHANNEL_INFO(chip,
			   HWMON_C_REGISTER_TZ | HWMON_C_UPDATE_INTERVAL),
	HWMON_CHANNEL_INFO(temp,
			   HWMON_T_INPUT | HWMON_T_MAX | HWMON_T_MAX_HYST),
	NULL
};

static const struct hwmon_ops nst175_hwmon_ops = {
	.is_visible = nst175_is_visible,
	.read = nst175_read,
	.write = nst175_write,
};

static const struct hwmon_chip_info nst175_chip_info = {
	.ops = &nst175_hwmon_ops,
	.info = nst175_info,
};

static bool nst175_is_writeable_reg(struct device *dev, unsigned int reg)
{
	return reg != NST175_REG_TEMP;
}

static bool nst175_is_volatile_reg(struct device *dev, unsigned int reg)
{
	return reg == NST175_REG_TEMP || reg == NST175_REG_CONF;
}

static const struct regmap_config nst175_regmap_config = {
	.reg_bits = 8,
	.val_bits = 16,
	.max_register = 0x03,
	.writeable_reg = nst175_is_writeable_reg,
	.volatile_reg = nst175_is_volatile_reg,
	.val_format_endian = REGMAP_ENDIAN_BIG,
	.cache_type = REGCACHE_RBTREE,
	.use_single_read = true,
	.use_single_write = true,
};

static void nst175_disable_regulator(void *data)
{
	struct nst175_data *nst175 = data;

	regulator_disable(nst175->vs);
}

static void nst175_remove(void *data)
{
	struct nst175_data *nst175 = data;
	struct i2c_client *client = nst175->client;

	i2c_smbus_write_byte_data(client, NST175_REG_CONF, nst175->orig_conf);
}

static const struct i2c_device_id nst175_ids[];

static int nst175_probe(struct i2c_client *client)
{
	struct device *dev = &client->dev;
	struct device *hwmon_dev;
	struct nst175_data *data;
	int status, err;
	enum nst175_type kind;

	if (client->dev.of_node)
		kind = (enum nst175_type)of_device_get_match_data(&client->dev);
	else
		kind = i2c_match_id(nst175_ids, client)->driver_data;

	if (!i2c_check_functionality(client->adapter,
			I2C_FUNC_SMBUS_BYTE_DATA | I2C_FUNC_SMBUS_WORD_DATA))
		return -EIO;

	data = devm_kzalloc(dev, sizeof(struct nst175_data), GFP_KERNEL);
	if (!data)
		return -ENOMEM;

	data->client = client;
	data->kind = kind;

	data->vs = devm_regulator_get(dev, "vs");
	if (IS_ERR(data->vs))
		return PTR_ERR(data->vs);

	data->regmap = devm_regmap_init_i2c(client, &nst175_regmap_config);
	if (IS_ERR(data->regmap))
		return PTR_ERR(data->regmap);

	data->params = &device_params[data->kind];

	/* Save default sample time and resolution*/
	data->sample_time = data->params->default_sample_time;
	data->resolution = data->params->default_resolution;

	/* Enable the power */
	err = regulator_enable(data->vs);
	if (err) {
		dev_err(dev, "failed to enable regulator: %d\n", err);
		return err;
	}

	err = devm_add_action_or_reset(dev, nst175_disable_regulator, data);
	if (err)
		return err;

	/* Cache original configuration */
	status = i2c_smbus_read_byte_data(client, NST175_REG_CONF);
	if (status < 0) {
		dev_dbg(dev, "Can't read config? %d\n", status);
		return status;
	}
	data->orig_conf = status;
	data->current_conf = status;

	err = nst175_write_config(data, data->params->set_mask,
				data->params->clr_mask);
	if (err)
		return err;

	err = devm_add_action_or_reset(dev, nst175_remove, data);
	if (err)
		return err;

	hwmon_dev = devm_hwmon_device_register_with_info(dev, client->name,
							 data, &nst175_chip_info,
							 NULL);
	if (IS_ERR(hwmon_dev))
		return PTR_ERR(hwmon_dev);

	dev_info(dev, "%s: sensor '%s'\n", dev_name(hwmon_dev), client->name);

	return 0;
}

static const struct i2c_device_id nst175_ids[] = {
	{ "nst175", nst175, },
	{ "nst175h-qmsr", nst175, },
	{ /* LIST END */ }
};
MODULE_DEVICE_TABLE(i2c, nst175_ids);

static const struct of_device_id __maybe_unused nst175_of_match[] = {
	{
		.compatible = "nsi,nst175",
		.data = (void *)nst175
	},
	{
		.compatible = "nsi,nst175h-qmsr", 
		.data = (void *)nst175
	},
	{ },
};
MODULE_DEVICE_TABLE(of, nst175_of_match);

#ifdef CONFIG_PM
static int nst175_suspend(struct device *dev)
{
	int status;
	struct i2c_client *client = to_i2c_client(dev);

	status = i2c_smbus_read_byte_data(client, NST175_REG_CONF);
	if (status < 0) {
		dev_dbg(&client->dev, "Can't read config? %d\n", status);
		return status;
	}
	status = status | NST175_SHUTDOWN;
	i2c_smbus_write_byte_data(client, NST175_REG_CONF, status);
	return 0;
}

static int nst175_resume(struct device *dev)
{
	int status;
	struct i2c_client *client = to_i2c_client(dev);

	status = i2c_smbus_read_byte_data(client, NST175_REG_CONF);
	if (status < 0) {
		dev_dbg(&client->dev, "Can't read config? %d\n", status);
		return status;
	}
	status = status & ~NST175_SHUTDOWN;
	i2c_smbus_write_byte_data(client, NST175_REG_CONF, status);
	return 0;
}

static const struct dev_pm_ops nst175_dev_pm_ops = {
	.suspend	= nst175_suspend,
	.resume		= nst175_resume,
};
#define NST175_DEV_PM_OPS (&nst175_dev_pm_ops)
#else
#define NST175_DEV_PM_OPS NULL
#endif /* CONFIG_PM */

static struct i2c_driver nst175_driver = {
	.class		= I2C_CLASS_HWMON,
	.driver = {
		.name	= "nst175",
		.of_match_table = of_match_ptr(nst175_of_match),
		.pm	= NST175_DEV_PM_OPS,
	},
	.probe_new	= nst175_probe,
	.id_table	= nst175_ids,
	.address_list	= normal_i2c,

module_i2c_driver(nst175_driver);

MODULE_AUTHOR("dll");
MODULE_DESCRIPTION("NST175 driver");
MODULE_LICENSE("GPL");
```

### 1.3.2. nst175.h
```
#include <linux/kernel.h>

/* straight from the datasheet */
#define LM75_TEMP_MIN (-55000)
#define LM75_TEMP_MAX 125000
#define LM75_SHUTDOWN 0x01

/*
 * TEMP: 0.001C/bit (-55C to +125C)
 * REG: (0.5C/bit, two's complement) << 7
 */
static inline u16 LM75_TEMP_TO_REG(long temp)
{
	int ntemp = clamp_val(temp, LM75_TEMP_MIN, LM75_TEMP_MAX);

	ntemp += (ntemp < 0 ? -250 : 250);
	return (u16)((ntemp / 500) << 7);
}

static inline int LM75_TEMP_FROM_REG(u16 reg)
{
	/*
	 * use integer division instead of equivalent right shift to
	 * guarantee arithmetic shift and preserve the sign
	 */
	return ((s16)reg / 128) * 500;
}
```
